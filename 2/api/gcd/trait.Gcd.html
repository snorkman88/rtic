source

fn gcd(self, other: Self) -> Self

Determine greatest common divisor using gcd_binary.

Examples
use gcd::Gcd;

assert_eq!(0, 0u8.gcd(0));
assert_eq!(10, 10u8.gcd(0));
assert_eq!(10, 0u8.gcd(10));
assert_eq!(10, 10u8.gcd(20));
assert_eq!(44, 2024u32.gcd(748));
